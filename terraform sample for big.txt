*using variables in tf

~ main.tf
resource "local_file" "sd" {
filename= var.filename
content=var.content
file_permission="0755" #default 0777
}
~ variable.tf
variable "filename" {
default="/home/rhyme/demo/test.txt"
type = string
{
variable "content" {
type=string
default="welcome to  my home"
}


terraform init
terraform plan
terraform apply
